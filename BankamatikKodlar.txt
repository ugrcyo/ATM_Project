        static void para_cekim(int miktar)
        {
            int a = 0;
            do
            {

                try
                {
                    if (miktar > para)
                    {
                        Console.WriteLine("Yetersiz bakiye");
                        //ana_menu_cikis();
                    }
                    else if (para >= miktar)
                    {
                        int bakiye;
                        bakiye = para - miktar;
                        para = bakiye;
                        Console.WriteLine("Kalan bakiye: {0}", bakiye);
                        
                        //ana_menu_cikis();
                    }
                }
                catch (Exception)
                {
                    a++;
                    Console.WriteLine("Yanlış tuşlama.Tekrar giriniz.");

                }
            } while (a > 0);


        }
        static void kredi_karti()
